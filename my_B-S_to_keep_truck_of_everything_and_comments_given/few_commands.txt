/opt/homebrew/opt/postgresql@16/bin/psql pgirl -c "
SELECT table_name, (xpath('/row/cnt/text()',
  query_to_xml(format('SELECT COUNT(*) AS cnt FROM %I', table_name), false, true, ''))
)[1]::text::int AS row_count
FROM information_schema.tables
WHERE table_schema = 'public' AND table_type = 'BASE TABLE'
ORDER BY table_name;"

ore / Data:

biopython (Bio) — NCBI Entrez queries, SeqIO, sequence alignment (PairwiseAligner)
Phylogenetics:

dendropy — synthetic phylogenetic tree generation
Geography:

pycountry — country name normalisation/matching
Web Search / API:

ddgs — DuckDuckGo search client (fallback for epidemiological data discovery)
requests — HTTP requests (UniProt, PubMed, PubTator API calls)

ollama (local, via HTTP — no pip package needed, just the Ollama app + model qwen2.5:14b)
Groq API (via urllib.request, needs GROQ_API_KEY env var — no pip package)
External tools (not pip packages):

PostgreSQL 16 — database (postgresql@16)
Nextclade CLI — genomic analysis (called via subprocess)
BLAST+ — taxonomic classification (called via subprocess)

runing the entire pipeline

cd /path/to/genomic_intelligence_workflow

# 1. Build the pgirl conda environment and verify packages
./nextflow run setup.nf

# 2. Update the PostgreSQL reference database (live run, not dry-run)
./nextflow run db_update.nf \
  --db_url postgresql://localhost:5432/pgirl \
  --sources all

# 3. Run the full end-to-end sample analysis pipeline
./nextflow run main.nf \
  --input_fasta /path/to/my_samples.fasta \
  --input_metadata /path/to/my_metadata.csv \
  --outdir /path/to/results \
  --db_url postgresql://localhost:5432/pgirl \
  --use_llm true

#set up LLM.

# Create a personal bin directory
mkdir -p ~/.local/bin
export PATH="$HOME/.local/bin:$PATH"

# Download Ollama for Linux (AMD64). For ARM64, replace amd64 with arm64.
curl -L https://ollama.com/download/ollama-linux-amd64 -o ~/.local/bin/ollama
chmod +x ~/.local/bin/ollama

# Verify
ollama --version

# 2. Start Ollama in background
ollama serve &

#Pull the model
ollama pull qwen2.5:14b

# 5. Stop Ollama
kill %1
